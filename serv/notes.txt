Revisiting the above
In particular, don't bother with ip addr add


* sudo apt-get install hostapd

*edit /etc/hostapd/hostapd.conf
interface=wlan0
driver=nl80211
beacon_int=100
hw_mode=g
ieee80211n=1
wme_enabled=1
country_code=US
ssid=vikingbot
wpa_passphrase=main400hs
ieee80211d=1
channel=3
wpa=2
wpa_key_mgmt=WPA-PSK
wpa_pairwise=TKIP
rsn_pairwise=CCMP
auth_algs=1
macaddr_acl=0
ignore_broadcast_ssid=0
#logger_syslog=-1
#logger_syslog_level=2
#logger_stdout=-1
#logger_stdout_level=2

* edit /etc/default/hostapd
DAEMON_CONF="/etc/hostapd/hostapd.conf"

* sudo apt-get install dnsmasq

* edit /etc/dnsmasq.conf
interface=wlan0
domain-needed
bogus-priv
dhcp-range=192.168.17.50,192.168.17.250,12h

* edit /etc/wpa_supplicant/wpa_supplicant.conf
remove existing networks (RickB, WMS-STEM) and add
network={
	ssid="vikingbot"
	psk="main400hs"
	key_mgmt=WPA-PSK
}

* edit /etc/dhcpcd.conf
interface wlan0
static ip_address=192.168.17.1
static routers=192.168.17.1


* sudo systemctl enable hostapd
* sudo systemctl enable dnsmasq
